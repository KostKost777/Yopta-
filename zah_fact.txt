улица Факториальная(дом x)
{
    если (x == 1) развернитесь
    {
        развернитесь_на 1;
    }
    после дом res поверните_на Факториальная(x - 1);
    развернитесь_на res * x;
}

маршрут_построен
{
    после дом x поверните_на 0;
    впереди x;
    крутой_поворот_на Факториальная(x);
}
